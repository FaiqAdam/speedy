    def construct_apache_touch_file
      if @app_name.empty?
        @touch_file = "no_file"
      else
        @touch_file = @app_name.include?('.') ? @app_name.split(".").shift : @app_name
      end
      return File.join(@@site_avaliable_dir_path, touch_file)
    end
